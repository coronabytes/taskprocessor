# Queues

{prefix}:queue:{queue}:deadletter	LIST

# Batches

{prefix}:batch:{batch}				HASH
  id			string
  start			long
  end			long?
  state			string
  done			long
  canceled		long
  failed		long
  duration		double
  tenant		string
  scope			string
  total			long
  remaining		long

{prefix}:batch:{batch}:continuations LIST
  {task}

{prefix}:batches:{tenant}			SORTED SET
  member: {batch}
  score: unix timestamp (seconds) of creation

{prefix}:batches:cleanup			SORTED SET
  member: {batch}
  score: unix timestamp (seconds) of eviction

# Schedules

{prefix}:schedule:{schedule}		HASH
  id		string
  timezone	string
  data		byte[]
  topic		string
  queue		string
  scope		string
  tenant	string
  retries	int
  cron		string
  next		long
  unique	bool
  expire	long

{prefix}:schedules					SORTED SET
  member: {schedule}
  score: unix timestamp (seconds) of next execution

{prefix}:schedules:{tenant}			SORTED SET
  member: {schedule}
  score: 0

{prefix}:task:{task}				HASH
  id			string
  schedule		string?
  batch			string?
  tenant		string
  data			byte[]
  queue			string?
  retries		int
  continuation	bool?
